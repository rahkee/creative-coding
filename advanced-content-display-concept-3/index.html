// JavaScript code with syntax highlighting
function calculateArea(radius) {
    const PI = 3.14159;
    return PI * radius ** 2;
}

const area = calculateArea(5);
console.log(`The area is ${area}`);
# Python example
def fibonacci(n):
    """Generate Fibonacci sequence up to n."""
    a, b = 0, 1
    while a < n:
        yield a
        a, b = b, a + b

for num in fibonacci(100):
    print(num)
